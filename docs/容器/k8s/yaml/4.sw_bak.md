==配合docker镜像使用==
``` yaml
apiVersion: batch/v1
kind: CronJob
metadata:
  name: sw-bak-cronjob
  namespace: sw-bak
spec:
  schedule: "0 8 * * 6"
  jobTemplate:
    spec:
      template:
        metadata:
          labels:
            app: redhat-container
        spec:
          containers:
          - name: sw-bak-cron
            image: localhost:30000/redhat/ubi8:ty
            volumeMounts:
            - name: local-volume
              mountPath: /home/
            command: ["/bin/bash", "-c"]
            args:
            - "bash -c 'bash /home/h3c.sh && bash /home/hw.sh'"
          volumes:
          - name: local-volume
            hostPath:
              path: /home/sw/
          restartPolicy: Never
  successfulJobsHistoryLimit: 1
  failedJobsHistoryLimit: 1
```
